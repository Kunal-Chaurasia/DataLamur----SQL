SELECT concat(p1.topping_name, ',', p2.topping_name, ',', p3.topping_name) as pizza,
p1.ingredient_cost + p2.ingredient_cost + p3.ingredient_cost as total_cost FROM pizza_toppings as p1
inner join pizza_toppings as p2 on p1.topping_name < p2.topping_name
inner join pizza_toppings as p3 on p2.topping_name < p3.topping_name
order by total_cost desc, pizza;
