with order_counts as (SELECT count(order_id) as total_items FROM orders)
select
  CASE
    when order_id % 2 != 0 and order_id != total_items then order_id + 1
    when order_id % 2 != 0 and order_id = total_items then order_id
    else order_id - 1
  end as corrected_order_id,
  item
from orders
cross join order_counts
order by corrected_order_id;
